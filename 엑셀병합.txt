Option Explicit

' ===== 설정값 =====
Const OPT_SKIP_HEADER As Boolean = True      ' 첫 파일 헤더만 유지, 이후 파일은 1행(헤더) 건너뛰기
Const OPT_APPEND_MODE As Boolean = True      ' 같은 시트명은 한 시트로 이어붙이기
Const OPT_STRICT_HEADER As Boolean = True    ' 헤더가 다르면 중단/알림

' 취합 대상 시트명
Const SHEET_DANJI As String = "단지정보"
Const SHEET_GROUP As String = "전용면적,구조로 그룹핑"

' 메인: 각 구별 엑셀들의 두 시트를 각각 통합
Public Sub MergeSigunguXlsx()
    Dim f As FileDialog, folder As String
    Set f = Application.FileDialog(msoFileDialogFolderPicker)
    f.Title = "시군구 엑셀들이 들어있는 폴더 선택"
    If f.Show <> -1 Then Exit Sub
    folder = f.SelectedItems(1)

    Dim t0 As Double: t0 = Timer
    Dim prevCalc As XlCalculation: prevCalc = Application.Calculation

    On Error GoTo FINALLY
    Application.ScreenUpdating = False
    Application.EnableEvents = False
    Application.DisplayAlerts = False
    Application.Calculation = xlCalculationManual

    Dim outWB As Workbook: Set outWB = Workbooks.Add
    Dim destDanji As Worksheet: Set destDanji = GetOrCreateDestSheet(outWB, SHEET_DANJI)
    Dim destGroup As Worksheet: Set destGroup = GetOrCreateDestSheet(outWB, SHEET_GROUP)

    ' 시트별 기준 헤더 보관
    Dim hdrDict As Object: Set hdrDict = CreateObject("Scripting.Dictionary")
    ' key: 시트명, val: 2차원 Variant (1행 헤더)

    Dim fname As String: fname = Dir(folder & "\*.xls*") ' xlsx/xlsm/xlsb/xls
    Dim inWB As Workbook, ws As Worksheet
    Dim lastCol As Long, lastRow As Long
    Dim firstDataRow As Long, destRow As Long
    Dim curHdr As Variant

    ' ▼ 오류 추적용
    Dim errFile As String
    Dim errSheet As String
    Dim errCol As Long
    Dim errRow As Long
    Dim inWBOpen As Boolean: inWBOpen = False

    Do While fname <> ""
        errFile = fname
        errSheet = ""
        errCol = 0
        errRow = 0
        inWBOpen = False
        Set inWB = Workbooks.Open(folder & "\" & fname, ReadOnly:=True)
        inWBOpen = True
        Application.StatusBar = "병합: " & fname

        For Each ws In inWB.Worksheets
            If ws.Name = SHEET_DANJI Or ws.Name = SHEET_GROUP Then
                errSheet = ws.Name
                If WorksheetHasData(ws) Then
                    lastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
                    If lastCol < 1 Then GoTo SkipSheet

                    ' 소스 헤더(1행)
                    curHdr = ws.Cells(1, 1).Resize(1, lastCol).Value2

                    ' 대상/기준시트 선택
                    Dim dest As Worksheet
                    If ws.Name = SHEET_DANJI Then
                        Set dest = destDanji
                    Else
                        Set dest = destGroup
                    End If

                    ' 기준 헤더 세팅(첫 파일)
                    If OPT_APPEND_MODE Then
                        If Not hdrDict.Exists(ws.Name) Or GetLastRow(dest) = 0 Then
                            dest.Cells.Clear
                            dest.Cells(1, 1).Resize(1, lastCol).Value2 = curHdr
                            hdrDict(ws.Name) = dest.Cells(1, 1).Resize(1, lastCol).Value2
                        End If

                        ' 엄격 모드: 기준 헤더와 비교(순서/개수/텍스트 동일)
                        If OPT_STRICT_HEADER Then
                            If Not HeaderEqual(hdrDict(ws.Name), curHdr) Then
                                inWB.Close False
                                Err.Raise vbObjectError + 777, , _
                                      "헤더 불일치: 시트 [" & ws.Name & "]" & vbCrLf & _
                                      "- 기준: " & Join1D(hdrDict(ws.Name)) & vbCrLf & _
                                      "- 발견: " & Join1D(curHdr) & vbCrLf & _
                                      "파일: " & fname
                            End If
                        End If

                        ' === 데이터 복사 ===
                        firstDataRow = IIf(OPT_SKIP_HEADER, 2, 1)
                        lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

                        If lastRow >= firstDataRow Then
                            Dim rowsCnt As Long: rowsCnt = lastRow - firstDataRow + 1
                            destRow = dest.Cells(dest.Rows.Count, 1).End(xlUp).Row + 1

                            Dim c As Long, headRaw As String, headKey As String
                            For c = 1 To lastCol
                                errCol = c
                                headRaw = CStr(ws.Cells(1, c).value)
                                headKey = CanonKeyForClassify(headRaw)

                                Dim rngDest As Range
                                Set rngDest = dest.Cells(destRow, c).Resize(rowsCnt, 1)

                                If ws.Name = SHEET_DANJI Then
                                    ' 단지정보: 숫자/문자 컬럼을 구분하여 복사
                                    If IsNumericHeaderName(headKey) Then
                                        ' --- 숫자 컬럼 ---
                                        Dim arrNum As Variant: ReDim arrNum(1 To rowsCnt, 1 To 1)
                                        Dim r As Long, s As String
                                        For r = 1 To rowsCnt
                                            errRow = firstDataRow + (r - 1)
                                            s = ws.Cells(errRow, c).Text
                                            s = CleanNumericStr(s)
                                            If s = "" Then
                                                arrNum(r, 1) = vbNullString
                                            ElseIf IsNumeric(s) Then
                                                arrNum(r, 1) = CDbl(s)
                                            Else
                                                arrNum(r, 1) = s ' 숫자화 불가 시 문자열 보존
                                            End If
                                        Next r
                                        rngDest.NumberFormat = "General"
                                        rngDest.value = arrNum
                                    Else
                                        ' --- 텍스트 컬럼(.Text) ---
                                        Dim arrTxt As Variant: ReDim arrTxt(1 To rowsCnt, 1 To 1)
                                        Dim k As Long
                                        For k = 1 To rowsCnt
                                            errRow = firstDataRow + (k - 1)
                                            arrTxt(k, 1) = ws.Cells(errRow, c).Text
                                        Next k
                                        rngDest.NumberFormat = "@"
                                        rngDest.value = arrTxt
                                    End If
                                Else
                                    ' 전용면적,구조로 그룹핑: 전체를 안전하게 텍스트로(.Text) 복사
                                    Dim arrTxt2 As Variant: ReDim arrTxt2(1 To rowsCnt, 1 To 1)
                                    Dim t As Long
                                    For t = 1 To rowsCnt
                                        errRow = firstDataRow + (t - 1)
                                        arrTxt2(t, 1) = ws.Cells(errRow, c).Text
                                    Next t
                                    rngDest.NumberFormat = "@"
                                    rngDest.value = arrTxt2
                                End If
                            Next c
                        End If
                        ' ====================
                    Else
                        ' append 모드가 아니면: 해당 시트만 복제(중복명 방지)
                        ws.Copy After:=outWB.Worksheets(outWB.Worksheets.Count)
                        On Error Resume Next
                        outWB.Worksheets(outWB.Worksheets.Count).Name = SafeSheetName(ws.Name & " (" & BaseName(fname) & ")")
                        On Error GoTo 0
                    End If
                End If
            End If
SkipSheet:
        Next ws

        inWBOpen = False
        inWB.Close False
        fname = Dir()
    Loop

    ' 저장
    Dim outPath As String
    outPath = folder & "\단지정보_내려받기_전국_합본_" & Format(Date, "yyyy-mm-dd") & ".xlsx"
    outWB.SaveAs Filename:=outPath, FileFormat:=xlOpenXMLWorkbook

    MsgBox "완료! " & vbCrLf & outPath & vbCrLf & _
           "소요: " & Format(Timer - t0, "0.0") & "초", vbInformation

FINALLY:
    Application.StatusBar = False
    Application.DisplayAlerts = True
    Application.Calculation = prevCalc
    Application.EnableEvents = True
    Application.ScreenUpdating = True

    ' ▼ 오류 시 열려 있는 inWB 닫기
    If inWBOpen Then
        On Error Resume Next
        inWB.Close False
        On Error GoTo 0
    End If

    If Err.Number <> 0 Then
        Dim msg As String
        msg = "오류가 발생하여 중단되었습니다." & vbCrLf & vbCrLf
        msg = msg & "파일  : " & errFile & vbCrLf
        If errSheet <> "" Then msg = msg & "시트  : " & errSheet & vbCrLf
        If errCol > 0 Then msg = msg & "열    : " & errCol & vbCrLf
        If errRow > 0 Then msg = msg & "행    : " & errRow & vbCrLf
        msg = msg & vbCrLf & "오류내용: " & Err.Description
        MsgBox msg, vbCritical
    End If
End Sub

' ------- 헬퍼들 -------

Private Function WorksheetHasData(ws As Worksheet) As Boolean
    On Error Resume Next
    WorksheetHasData = (WorksheetFunction.CountA(ws.UsedRange) > 0)
End Function

Private Function GetOrCreateDestSheet(wb As Workbook, sName As String) As Worksheet
    Dim nm As String: nm = Left(sName, 31)
    On Error Resume Next
    Set GetOrCreateDestSheet = wb.Worksheets(nm)
    On Error GoTo 0
    If GetOrCreateDestSheet Is Nothing Then
        Set GetOrCreateDestSheet = wb.Worksheets.Add(After:=wb.Worksheets(wb.Worksheets.Count))
        GetOrCreateDestSheet.Name = nm
    End If
End Function

Private Function HeaderEqual(h1 As Variant, h2 As Variant) As Boolean
    Dim c1 As Long, c2 As Long, i As Long
    On Error Resume Next
    c1 = UBound(h1, 2): c2 = UBound(h2, 2)
    On Error GoTo 0
    If c1 <> c2 Then HeaderEqual = False: Exit Function
    For i = 1 To c1
        If CStr(h1(1, i)) <> CStr(h2(1, i)) Then HeaderEqual = False: Exit Function
    Next i
    HeaderEqual = True
End Function

Private Function GetLastRow(ws As Worksheet) As Long
    GetLastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row
    If ws.Cells(1, 1).value = "" And GetLastRow = 1 Then GetLastRow = 0
End Function

Private Function GetLastCol(ws As Worksheet) As Long
    GetLastCol = ws.Cells(1, ws.Columns.Count).End(xlToLeft).Column
End Function

Private Function BaseName(ByVal fn As String) As String
    Dim p As Long: p = InStrRev(fn, ".")
    If p > 0 Then BaseName = Left(fn, p - 1) Else BaseName = fn
End Function

Private Function Join1D(arr As Variant) As String
    Dim i As Long, n As Long
    On Error Resume Next
    n = UBound(arr, 2)
    On Error GoTo 0
    If n = 0 Then Join1D = "": Exit Function
    Dim tmp() As String: ReDim tmp(1 To n)
    For i = 1 To n
        tmp(i) = CStr(arr(1, i))
    Next i
    Join1D = Join(tmp, " | ")
End Function

Private Function SafeSheetName(ByVal nm As String) As String
    Dim badChars As Variant: badChars = Array(":", "\", "/", "?", "*", "[", "]")
    Dim i As Long
    For i = LBound(badChars) To UBound(badChars)
        nm = Replace(nm, badChars(i), " ")
    Next i
    If Len(nm) > 31 Then nm = Left(nm, 31)
    SafeSheetName = nm
End Function

' === 분류/정규화 (단지정보 전용) ===

' 분류용 캐논 키: NBSP/공백/괄호 단위 제거, 소문자
Private Function CanonKeyForClassify(ByVal s As String) As String
    s = Replace(s, ChrW(160), " ")
    s = Replace(s, "（", "("): s = Replace(s, "）", ")")
    s = Trim(s)
    If InStr(s, "(") > 0 Then s = Trim(Split(s, "(")(0)) ' "전용면적(㎡)" -> "전용면적"
    s = Replace(s, " ", "")
    CanonKeyForClassify = LCase$(s)
End Function

' 숫자 컬럼인지 여부 (※ 평형명 제외)
Private Function IsNumericHeaderName(ByVal canonKey As String) As Boolean
    Select Case canonKey
        Case "총세대수", "공급면적", "전용면적", "구조", "방수", "욕실수", "평형별세대수", "평형별 세대수"
            IsNumericHeaderName = True
        Case Else
            IsNumericHeaderName = False
    End Select
End Function

' 숫자 변환 전 문자열 정리(단지정보 숫자열 대상)
Private Function CleanNumericStr(ByVal s As String) As String
    s = Replace(s, ChrW(160), " ")
    s = Replace(s, ",", "")
    s = Replace(s, "㎡", "")
    s = Replace(s, "m²", "")
    s = Replace(s, vbTab, "")
    s = Replace(Replace(s, vbCr, ""), vbLf, "")
    s = Trim(s)
    CleanNumericStr = s
End Function
